01. if문

조건식 결과에 따라 무슨 실행을 할 것인지 결정하는 조건문

{
    //true, 1 이라 쓰면 true가 나오고, false, 0, undefined, null 을 쓰면 false가 나온다.
    //문자열 "0 ~", [0](배열), {a: 100}(객체) 를 쓰면 true가 나온다.
    if("조건식"){ //true(참 출력)
        document.write("실행되었습니다(true).");
    } else { //false(거짓 출력)
        document.write("실행되었습니다(false).");
    }
}
결과보기
실행되었습니다(true).

02. if문 생략

괄호를 생략하고 실행시킨다.

    //{} 괄호 생략
    if(num) document.write("실행되었습니다(true).");
    else document.write("실행되었습니다(false).");
}

03. 다중 if문

여러 if문을 실행시켜 좀 더 구체적인 결과를 얻어낼 때 사용한다.

{
    const num = 100;
    
    if( num == 90 ){ //여러 if문을 실행한다.
        document.write("실행되었습니다(num == 90).");
    } else if( num == 100 ){
        document.write("실행되었습니다(num == 100).");
    } else if( num == 110 ){
        document.write("실행되었습니다(num == 110).");
    } else {
        document.write("실행되었습니다(num == 값이 없음).");
    }
    
}
결과보기
실행되었습니다(num == 100).

06. swich문

if문과 용도는 비슷하나 if문은 만족하는 데이터가 여러 개일 경우에 주로 사용하고,
witch문은 여러 경우의 값 중 일치하는 데이터를 찾아 그에 해당하는 코드를 실행시킬 때 사용

{
    const num = 100;
    
    switch(num){
        case 100 :
            document.write("실행되었습니다(num==100).");
            break;
        case 100 :
            document.write("실행되었습니다(num==120).");
            break;
    
        case 100 :
            document.write("실행되었습니다(num==130).");
            break;
            default :
            document.write("실행되었습니다(num==값이 없음)."); //마지막은 break 가 필요 없다.
    }
}
결과보기 실행되었습니다(num==100).

10. 중첩 for문

for문 안에 for문을 중첩하여 사용

{   //두번 반복
    for( let i=1; i<2; i++){
        document.write(i + "실행");
        for(let j=1; j<=5; j++){
            document.write(j + "실행");
            //i 를 한 번 실행할 때 j를 다섯 번 반복 시켜준다.
        }
    }
}

11. break 문

현재 반복문 종료합니다.

for(let i = 1; i<=20; i++ ){
    document.write(i);
    if (i == 10){
        break;       //멈춤
    }
}

12. continue 문

현재 반복 종료하고 그 다음부터 출력합니다.

for(let i = 1; i<=20; i++){
    if(i == 10){
        continue;         //현재 반복에서 명령문의 실행을 종료하고 그다음부터 출력
    }
    document.write(i);
}
