//scrollIntoView() : 스크롤이 이동하는 동안의 움직임 효과를 설정할 수 있습니다.
//getAttribute() : 특정 요소 내에서 특정 한 속성값을 가져오는 메서드
document.querySelectorAll("#parallax__nav li a").forEach(li => {
    //이벤트 발생을 막는다.(behavior: "smooth"를 부여하기 위함.)
    li.addEventListener("click", (e) => {
        e.preventDefault();
        document.querySelector(li.getAttribute("href")).scrollIntoView({    //li = li a태그들 href = secton1,2,...//getAttribute 속성값을 가져온다.
            behavior: "smooth"
        });
    });
});
