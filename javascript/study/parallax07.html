js

p태그 내에 reveal이 부여된 글들을 span으로 묶을 것입니다. reveal이 부여된 것을 querySelectorAll로 모두 찾아서 forEach로 진행을 합니다. splitText에 텍스트를 저장하고, 텍스트를 span으로 묶어줍니다. 스크롤에대한 이벤트를 만듭니다. revealDelay는 각 html에 부여된 dataset-delay의 값입니다. 없는경우 undefined 보여주고, 그렇지않은 경우에는 정해진 딜레이에맞게 show. 

const reveal = document.querySelectorAll(".reveal");
reveal.forEach((el) => {
    let splitText = el.innerText;
    if (splitText) {
        el.innerHTML = "<span>" + splitText + "</span>";
    }
});

// reveal 클래스 자식에 글씨만 있다면, 가상으로 spa태그를 만들어서 넣어주세요! (6번의 내용 참고)
function scroll() {
    let scrollTop = window.scrollY;

    reveal.forEach(el => {
        let revealOffset = el.offsetTop + el.parentElement.offsetTop;
        let revealDelay = el.dataset.delay;

        // if (scrollTop > revealOffset - window.innerHeight) {
        //     el.classList.add("show");
        // }
        if (scrollTop >= revealOffset - window.innerHeight) {
            if (revealDelay == undefined) {
                el.classList.add("show");
            } else {
                setTimeout(() => {
                    el.classList.add("show");
                }, revealDelay)
            }
        }
    })

    document.querySelector(".scroll span").innerText = Math.round(scrollTop);
    requestAnimationFrame(scroll);
}
