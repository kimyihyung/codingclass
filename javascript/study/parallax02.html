js

preventDefault() : 요소를 클릭했을 때, 링크로 이동되지 않게 합니다.
=> scrollIntoView() : 특정 위치로 이동시켜주는 함수로 behavior: "smooth"을 부여하여 움직임을 부드럽게 하기위해.

getAttribute() : 특정 요소 내에서 특정 한 속성값을 가져오는 메서드

(scrollTop >= e.offsetTop - 10) = e.offsetTopdl이 scrollTop보다 수치가 커지는 오차가 발생. 10미만의 오차이지만, 10으로 부여. 탑과 section의 오차를 줄여줍니다. 

document.querySelectorAll("#parallax__dot a").forEach(el => {
    el.addEventListener("click", e => {
        e.preventDefault(); //링크를 차단

        document.querySelector(el.getAttribute("href")).scrollIntoView({ behavior: "smooth" })
    });
});

window.addEventListener("scroll", () => {
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop; //브라우저마다 지원여부가 다르기에 "||"을 적어 셋 중 한가지를 선택해 스크롤 값을 불어와서 작동하도록.

    document.querySelector("#parallax__info span").innerText = Math.floor(scrollTop);

    document.querySelectorAll(".content__item").forEach((e, i) => {
        if (scrollTop >= e.offsetTop - 10) {    //나중에 가면 e.offsetTopdl이 scrollTop보다 수치가 커지는 오차가 발생. 10미만의 오차이지만, 10으로 부여. 탑과 section의 오차를 줄이기.
            document.querySelectorAll("#parallax__dot li").forEach(li => {
                li.classList.remove("active");
            });
            document.querySelector("#parallax__dot li:nth-child(" + (i + 1) + ")").classList.add("active")
        };
    });
});
