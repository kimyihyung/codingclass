js

글자를 한글자씩 쪼개는 과정을 가집니다.split()을 이용해서 쪼갠 뒤에 글자들이 나눠지면 웹접근성에 안좋아지는 문제가 발생합니다. 글자들이 한 글자씩 나눠지기를 바라지만, 보여지는 것과 읽히는 것은 본래의 문장으로 잃혀야하는데 쪼개는 경우에 쪼개진 것으로 읽는 문제가 발생하기에 join()을 이용해서 접근성을 위해서 span에 aria-hidden="true"을 부여합니다. 스크롤에 대한 이벤트를 부여합니다. .split span에 show가 붙으면, index 이용하여 transition-delay효과 추가 

// 글씨 분리하기
// let text = document.querySelector("#section1 .content__item__desc");
// let splitText = text.innerText;
// let splitWrap = splitText.split('').join('</span><span>')
// text.innerHTML = splitWrap = "<span>" + splitWrap + "</span>";

// 클씨 분리(다중이) (하나만 바꾸는게 아니라 9개 전부를 forEach)
document.querySelectorAll(".split").forEach(text => {
    let splitText = text.innerText;
    let splitWrap = splitText.split('').join('</span><span aria-hidden="true">');   //접근성을 위한 aria-hidden
    splitWrap = "<span aria-hidden='true'>" + splitWrap + "</span>";
    text.innerHTML = splitWrap;
    text.setAttribute("aria-label", splitText);
});

function scroll() {
    let scrollTop = window.scrollY;

    document.querySelector("#parallax__info .scroll").innerText = Math.round(scrollTop);

    // CSS 클래스 추가 (비효율)
    // document.querySelectorAll(".content__item").forEach(item => {
    //     if (scrollTop >= item.offsetTop) {
    //         item.querySelector(".split").classList.add("show");
    //     };
    // });

    document.querySelectorAll(".content__item").forEach(item => {
        if (scrollTop >= item.offsetTop) {
            item.querySelectorAll(".split span").forEach((span, index) => {
                span.classList.add("show");

                span.style.transitionDelay = `${index * 50}ms`
            });
        }
    })

    requestAnimationFrame(scroll);

    //index에 있음. 시간에 따라서 순서대로 글자가 나타나게
    //첫번째 span ---> 0.01
    //두번째 span ---> 0.02
    //세번째 span ---> 0.03
    //네번째 span ---> 0.04
    //다섯번째 span ---> 0.05
    //여섯번째 span ---> 0.06
    //일곱번째 span ---> 0.07
}
