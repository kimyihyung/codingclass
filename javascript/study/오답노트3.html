02. 다음의 결괏값을 작성하시오.

{
    function myWeight() {
        let kcal, weight = 0;
    
        for (kcal = 0; kcal <= 8; kcal++) {
            weight += kcal;
            if (kcal == 5) {
                break;
            }
        }
        console.log(weight + kcal);
    }
    myWeight()
}

break는 끝나게 되고, continue는 그 부분만 건너뛰는 것인데 break이기에, weight는 0, 1, 3, 6, 10, 15으로 점차 증가하여 15가 됩니다. kcal은 5이기에 5를 더하여 답이 20이 됩니다.

정답 확인하기
20

03. 다음의 결괏값을 작성하시오.

{
    function party() {
        let join = 0;
        let people = ['철수', '민수', '상수', '변수', '무리수']
        let enable = 4;
    
        if (people.length <= 5) {
            enable = 3;
        } else {
            enable = 8;
        }
        while (join < people.length) {
            join++;
            if (join == enable) break;
        }
        console.log("오늘 파티 참가 가능한 인원수는 총 " + join + "명");
    }
    party();
}

people.length은 5이기 때문에 enable = 3;으로 3명이 됩니다.

정답 확인하기
"오늘 파티 참가 가능한 인원수는 총 3명"

04. 다음의 결괏값을 작성하시오.

{
    function func() {
        let i = 0, sum = 0, sel = 0;
    
        for (let sel = 1; sel <= 8; sel++) {
            if (sel < 4) break;
        }
        if (sel <= 4) {
            while (i <= 5) {
                i++;
                sum += i;
            }
        } else {
            while (i <= 3) {
                i++;
                sum += i;
            }
        }
        console.log(sum);
    }
    func()
}

sel은 4보다 작은 값에서 break가 됩니다. 그에 따라서 첫번째 if에서 4이하라는 조건에 맞고, while문을 실행합니다. (i <= 5)밑에 i++로 인해서 i는 1에서 6까지가 되며, 0+1=1, 1+2=3, 3+3=6, 6+4=10, 10+5=15, 15+6=21이 되며 답은 21이 됩니다.

정답 확인하기
21

{

07. 다음의 결괏값을 작성하시오.

{
    function good() {
        let list = [1, 2, 3, 4, 5, 6];
        let i = 0;
    
        for (let q = 0; q <= 23; q++) {
            if (q % 2 == 0) {
                i++
            } else {
                i--
            }
            if (q == 2) break;
            console.log(list[i]);
        }
    }
    good();
}

if (q == 2) break;에 의하여 2까지만 q안에 들어갑니다. 0에서 2를 대입하게 되는데, 0을 대입하면 처음 if에 부합하며, i++로 1이 됩니다. list[i]에서 1의 값은 2이며, 1은 조건에 맞지 않아서 i--로 인해서 0이되고, 배열 안에서 값은 1. 2는 브레이크로 멈추기에 2,1이 답.

정답 확인하기
2,1

08. 다음의 결괏값을 작성하시오.

{
    function solution(numbers) {
        let answer = 0;
        for (let i = 0; i <= 10; i++) {
            if (!numbers.includes(i)) {
                answer += i;
            }
        }
        if (answer <= 10) {
            answer = 0;
        } else {
            answer = 1;
        }
        return answer;
    }
    console.log(solution([1, 2, 3, 4, 5, 6, 7, 8, 0]));
}

includes로 numbers의 i안에서 포함되지 않은 값. 그것은 9, 10이며, 둘을 더하면 19이고 그것은 10을 넘기 때문에 1이 된다.

정답 확인하기
1
